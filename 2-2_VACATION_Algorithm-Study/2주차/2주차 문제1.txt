#include<stdio.h>
int order(int n,int r[],int s[]){
	int i,j;

	for(i=0;i<n;i++){
		if(r[i]>i)return 0;
		else{
		s[i]=r[i]+1;
		for(j=0;j<i;j++)
			if(s[i]<=s[j])s[j]++;}
	}
	return 1;
}

int main(){
	int n,wn,i,j;
scanf("%d",&wn);

for(i=0;i<wn;i++){
	scanf("%d",&n);
	int *r=(int*)malloc(sizeof(int)*n);
	int *s=(int*)malloc(sizeof(int)*n);
	for(j=0;j<n;j++)s[j]=0;
	for(j=0;j<n;j++)
	scanf("%d",&r[j]);
	if(order(n,r,s)){
		for(j=0;j<n;j++) printf("%d ",s[j]);
		printf("\n");
	}
	else {printf("IMPOSSIBLE");
	printf("\n");
	}
}

return 0;
}
